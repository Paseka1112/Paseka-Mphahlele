#P1(a)
import numpy as np
import matplotlib.pyplot as plt

def f(y):
    return y / (1 + y)

def forward_euler(y0, h):
    t = np.arange(0, 1+h, h)
    y = np.zeros(len(t))
    y[0] = y0
    for i in range(1, len(t)):
        y[i] = y[i-1] + h * f(y[i-1])
    return y

def trapezium(y0, h):
    t = np.arange(0, 1+h, h)
    y = np.zeros(len(t))
    y[0] = y0
    for i in range(1, len(t)):
        y[i] = y[i-1] + h/2 * (f(y[i-1]) + f(y[i-1] + h*f(y[i-1])))
    return y

def midpoint(y0, h):
    t = np.arange(0, 1+h, h)
    y = np.zeros(len(t))
    y[0] = y0
    for i in range(1, len(t)):
        y[i] = y[i-1] + h * f(y[i-1] + h/2 * f(y[i-1]))
    return y

h_vals = [0.0100, 0.0050, 0.0025]
y_true = 1.557145598997611
y0 = 1

print('Method\t\th\t\ty(1)\t\tError')

for h in h_vals:
    y_fe = forward_euler(y0, h)[-1]
    y_tm = trapezium(y0, h)[-1]
    y_mm = midpoint(y0, h)[-1]
    err_fe = abs(y_fe - y_true)
    err_tm = abs(y_tm - y_true)
    err_mm = abs(y_mm - y_true)
    print(f'Forward Euler\t{h:.4f}\t{y_fe:.12f}\t{err_fe:.12f}')
    print(f'Trapezium\t{h:.4f}\t{y_tm:.12f}\t{err_tm:.12f}')
    print(f'Midpoint\t{h:.4f}\t{y_mm:.12f}\t{err_mm:.12f}')
    print()

plt.figure()
for method, label in zip([forward_euler, trapezium, midpoint], ['Forward Euler', 'Trapezium', 'Midpoint']):
    errors = [abs(method(y0, h)[-1] - y_true) for h in h_vals]
    plt.loglog(h_vals, errors, label=label)
plt.xlabel('h')
plt.ylabel('Error')
plt.legend()
plt.show()

def implicit_trapezium(y0, h):
    t = np.arange(0, 1+h, h)
    y = np.zeros(len(t))
    y[0] = y0
    for i in range(1, len(t)):
        y[i] = (y[i-1] + h/2 * (f(y[i-1]) + f(y[i-1] + h*f(y[i])))) / (1 + h/2 * f(y[i]))
    return y

import numpy as np
import matplotlib.pyplot as plt
from scipy.integrate import solve_ivp

def dydt(t, y):
    return y**2 - y**3

delta = 0.001
y0 = delta

tspan = [0, 2/delta]

options = {'rtol': 1e-10}

sol = solve_ivp(dydt, tspan, [y0], method='RK45', **options)

print(f"Number of function evaluations: {sol.nfev}")
print(f"Number of steps: {sol.nfev - 1}")

import numpy as np
from scipy.integrate import solve_ivp
import matplotlib.pyplot as plt

def ode_system(x, y):
    u, v = y
    n = 1
    if x == 0:
        rhs = [-2 * v - (1/3) * x * u**n, v]
    else:
        rhs = [v, -2 * v - x * u**n]
    return rhs

x_start = 0
x_end = 10

u_initial = 1
v_initial = 0
initial_conditions = [u_initial, v_initial]

solution = solve_ivp(ode_system, [x_start, x_end], initial_conditions, dense_output=True)

x_values = np.linspace(x_start, x_end, 100)
y_values = solution.sol(x_values)

plt.plot(x_values, y_values[0], label='u(x)')
plt.plot(x_values, y_values[1], label='u\'(x)')
plt.xlabel('x')
plt.ylabel('y')
plt.legend()
plt.title('Solution of the ODE system')
plt.grid(True)
plt.show()

#P3(B)

import numpy as np
from scipy.integrate import solve_ivp
import matplotlib.pyplot as plt

def ode_system(x, y, n):
    u, v = y
    if x == 0:
        rhs = [-2 * v - (1/3) * x * u**n, v]
    else:
        rhs = [v, -2 * v - x * u**n]
    return rhs

x_start = 0
x_end = 10

u_initial = 1
v_initial = 0
initial_conditions = [u_initial, v_initial]

n_values = range(6)

solutions = []
for n in n_values:
    solution = solve_ivp(lambda x, y: ode_system(x, y, n), [x_start, x_end], initial_conditions, dense_output=True)
    solutions.append(solution.sol)

x_values = np.linspace(x_start, x_end, 100)

plt.figure(figsize=(10, 6))
for n, solution in zip(n_values, solutions):
    u_values = solution(x_values)[0]
    plt.plot(x_values, u_values, label=f'n = {n}')

plt.xlabel('x')
plt.ylabel('u(x)')
plt.ylim(-0.5, 1)
plt.legend()
plt.title('Solutions of the ODE for different values of n')
plt.grid(True)
plt.show()

def ssprk3(f, tspan, y0, N):
    h = (tspan[-1] - tspan[0]) / N
    w = y0
    t = tspan[0]
    
    for j in range(N):
        s1 = f(t, w)
        s2 = f(t + h / 2, w + h / 4 * s1)
        s3 = f(t + h, w - h / 6 * s1 + h / 6 * s2)
        
        y = w + h / 6 * (s1 + 4 * s2 + s3)
        t = tspan[0] + (j + 1) * h
        
        w = y
    
    return w
